<script setup>
import { ref, reactive, computed } from 'vue'
import { useRouter } from 'vue-router'
import axios from 'axios'
import RouteMap from '../components/RouteMap.vue'
 
const router = useRouter()
const loading = ref(false)
const errorMessage = ref('')
const searched = ref(false)
const schedules = ref([])
const selectedSchedule = ref(null)
const originPoint = ref(null)
const destPoint = ref(null)
const form = reactive({ originName: '', destName: '', departDate: '' })
const suggestions = reactive({ origin: [], dest: [] })
const geocoding = reactive({ origin: false, dest: false })
const debounceTimers = { origin: null, dest: null }
 
async function geocode(query) {
  const url = `https://nominatim.openstreetmap.org/search?q=${encodeURIComponent(query)}&format=json&limit=8&countrycodes=jp&accept-language=ja`
  const res = await fetch(url, { headers: { 'Accept-Language': 'ja' } })
  const results = await res.json()
  const seen = new Set()
  return results.filter(s => {
    const key = `${parseFloat(s.lat).toFixed(4)},${parseFloat(s.lon).toFixed(4)}`
    if (seen.has(key)) return false; seen.add(key); return true
  }).slice(0, 5).map(s => ({ ...s, display_name: shortName(s.display_name) }))
}
 
function shortName(full) {
  const parts = full.split(',').map(p => p.trim())
  const area = parts.slice(1, 4).filter(p => p && !/^\d/.test(p) && p !== '日本').slice(0, 2).join(', ')
  return area ? `${parts[0]}（${area}）` : parts[0]
}
 
function onOriginNameInput() {
  if (originPoint.value) originPoint.value = null
  clearTimeout(debounceTimers.origin)
  if (form.originName.length < 2) { suggestions.origin = []; return }
  debounceTimers.origin = setTimeout(async () => {
    geocoding.origin = true
    try { suggestions.origin = await geocode(form.originName) } finally { geocoding.origin = false }
  }, 400)
}
 
function onDestNameInput() {
  if (destPoint.value) destPoint.value = null
  clearTimeout(debounceTimers.dest)
  if (form.destName.length < 2) { suggestions.dest = []; return }
  debounceTimers.dest = setTimeout(async () => {
    geocoding.dest = true
    try { suggestions.dest = await geocode(form.destName) } finally { geocoding.dest = false }
  }, 400)
}
 
function selectSuggestion(type, s) {
  const point = { lat: parseFloat(s.lat), lng: parseFloat(s.lon), name: s.display_name }
  if (type === 'origin') { originPoint.value = point; form.originName = s.display_name; suggestions.origin = [] }
  else { destPoint.value = point; form.destName = s.display_name; suggestions.dest = [] }
}
 
function hideDropdown(type, delay) { setTimeout(() => { suggestions[type] = [] }, delay) }
function resetPoints() { originPoint.value = null; destPoint.value = null; form.originName = ''; form.destName = '' }
 
const RADIUS = 0.3
function toBBox(p) { return { latMin: p.lat - RADIUS, latMax: p.lat + RADIUS, lngMin: p.lng - RADIUS, lngMax: p.lng + RADIUS } }
 
async function handleSearch() {
  loading.value = true; errorMessage.value = ''; searched.value = false; selectedSchedule.value = null
  try {
    const params = {}
    if (originPoint.value) { const b = toBBox(originPoint.value); params.origin_lat_min = b.latMin; params.origin_lat_max = b.latMax; params.origin_lng_min = b.lngMin; params.origin_lng_max = b.lngMax }
    if (destPoint.value) { const b = toBBox(destPoint.value); params.dest_lat_min = b.latMin; params.dest_lat_max = b.latMax; params.dest_lng_min = b.lngMin; params.dest_lng_max = b.lngMax }
    if (form.departDate) { params.depart_at_from = new Date(form.departDate + 'T00:00:00').toISOString(); params.depart_at_to = new Date(form.departDate + 'T23:59:59').toISOString() }
    const res = await axios.get('/api/v1/schedules/search', { params })
    schedules.value = res.data.schedules || []; searched.value = true
  } catch { errorMessage.value = 'スケジュールの検索に失敗しました。' }
  finally { loading.value = false }
}
 
const selectedOrigin = computed(() => selectedSchedule.value ? { lat: selectedSchedule.value.origin_lat, lng: selectedSchedule.value.origin_lng } : null)
const selectedDest = computed(() => selectedSchedule.value ? { lat: selectedSchedule.value.dest_lat, lng: selectedSchedule.value.dest_lng } : null)
const mapBounds = computed(() => { const s = selectedSchedule.value; return s ? [[s.origin_lat, s.origin_lng], [s.dest_lat, s.dest_lng]] : null })
 
function selectSchedule(s) { selectedSchedule.value = s }
function goToBooking(s) { router.push(`/shipper/bookings/new?schedule_id=${s.id}`) }
 
function formatDate(d) {
  if (!d) return '-'
  return new Date(d).toLocaleString('ja-JP', { year: 'numeric', month: '2-digit', day: '2-digit', hour: '2-digit', minute: '2-digit' })
}
 
function statusLabel(s) { return { open: '受付中', full: '満載', departed: '出発済', arrived: '到着済' }[s] || s }
function statusClass(s) {
  return { open: 'bg-green-100 text-green-800', full: 'bg-orange-100 text-orange-700', departed: 'bg-gray-100 text-gray-600', arrived: 'bg-blue-100 text-blue-800' }[s] || 'bg-gray-100 text-gray-600'
}
</script>
